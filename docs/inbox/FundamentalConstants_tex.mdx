\section{Theoretical Foundation}
%==============================================================================

\subsection{Fundamental Constants in Atomic Physics}

\begin{axiom}[CODATA Standards]
Physical constants used in precision atomic physics calculations must conform to CODATA recommended values to ensure reproducibility and accuracy across different implementations.
\end{axiom}

\begin{definition}[Bohr Magneton]
The Bohr magneton is the natural unit of magnetic moment for an electron:
\begin{equation}
\mu_B = \frac{e\hbar}{2m_e} \approx 9.274 \times 10^{-24} \text{ J/T}
\end{equation}
\end{definition}

\begin{definition}[Electron g-factor]
The electron spin g-factor relates the magnetic moment to the spin angular momentum:
\begin{equation}
\vec{\mu}_s = -g_s \frac{\mu_B}{\hbar} \vec{S}
\end{equation}
where $g_s \approx 2.002319$ (including QED corrections).
\end{definition}

\begin{theorem}[Zeeman Energy]
The magnetic interaction energy for an electron in field $\vec{B}$ is:
\begin{equation}
H_Z = -\vec{\mu} \cdot \vec{B} = \mu_B (g_L L_z + g_S S_z) B
\end{equation}
where $g_L = 1$ for orbital and $g_S \approx 2$ for spin contributions.
\end{theorem}

\subsection{Atomic Units and Conversions}

\begin{definition}[Bohr Radius]
The characteristic length scale of atomic physics:
\begin{equation}
a_0 = \frac{4\pi\epsilon_0\hbar^2}{m_e e^2} \approx 5.292 \times 10^{-11} \text{ m}
\end{equation}
\end{definition}

\begin{definition}[Atomic Mass Unit]
The unified atomic mass unit:
\begin{equation}
1 \text{ amu} = \frac{m(^{12}\text{C})}{12} \approx 1.661 \times 10^{-27} \text{ kg}
\end{equation}
\end{definition}

\textit{Natural unit for atomic magnetic moments, essential for Zeeman effect calculations.}

\section{Physical Applications}
%==============================================================================

\subsection{Zeeman Effect Calculations}

The magnetic field interaction uses $\mu_B$ and $g_s$:
\begin{equation}
\Delta E = \mu_B B (m_L g_L + m_S g_S + m_I g_I)
\end{equation}

\subsection{Doppler Width}

Temperature-dependent line broadening uses $k_B$ and atomic mass:
\begin{equation}
\Delta\nu_D = \nu_0 \sqrt{\frac{8 k_B T \ln 2}{m c^2}}
\end{equation}

\subsection{Dipole Matrix Elements}

The transition dipole moment involves $a_0$ and $e$:
\begin{equation}
d = e \langle \psi_f | \vec{r} | \psi_i \rangle \sim e \cdot a_0
\end{equation}

\section{Summary}
%==============================================================================

This module provides the essential physical constants for atomic physics calculations:

\begin{enumerate}
\item $S = 1/2$: Electron spin quantum number
\item $g_s \approx 2$: Electron spin g-factor
\item $\mu_B$: Bohr magneton for magnetic interactions
\item $k_B$: Boltzmann constant for thermal effects
\item $\epsilon_0, a_0$: Electromagnetic and atomic length scales
\item amu: Mass unit for velocity/Doppler calculations
\end{enumerate}

These constants are imported by other modules in the ElecSus package for computing atomic spectra.
