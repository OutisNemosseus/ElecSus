\section{Theoretical Foundation}
%==============================================================================

\subsection{Atomic Structure Parameters}

\begin{definition}[Nuclear Spin]
The nuclear spin $I$ determines the hyperfine structure:
\begin{equation}
F = |I - J|, |I - J| + 1, \ldots, I + J
\end{equation}
giving $(2\min(I,J) + 1)$ hyperfine levels for each fine structure state.
\end{definition}

\begin{definition}[Magnetic Dipole Hyperfine Constant]
The $A$ coefficient characterizes the $\vec{I} \cdot \vec{J}$ interaction:
\begin{equation}
A = \frac{\mu_0 g_I \mu_N \mu_B}{4\pi} \frac{1}{I J} \langle r^{-3} \rangle
\end{equation}
\end{definition}

\begin{definition}[Electric Quadrupole Hyperfine Constant]
The $B$ coefficient (for $I \geq 1$, $J \geq 1$):
\begin{equation}
B = e Q \langle \frac{\partial^2 V}{\partial z^2} \rangle_0
\end{equation}
where $Q$ is the nuclear quadrupole moment.
\end{definition}

\begin{theorem}[Hyperfine Energy Levels]
The hyperfine energy is:
\begin{equation}
E_{HFS} = \frac{A \cdot K}{2} + B \frac{\frac{3}{2}K(K+1) - 2I(I+1)J(J+1)}{4I(2I-1)J(2J-1)}
\end{equation}
where $K = F(F+1) - I(I+1) - J(J+1)$.
\end{theorem}

\subsection{Transition Properties}

\begin{definition}[Natural Linewidth]
The spontaneous decay rate determines the natural linewidth:
\begin{equation}
\Gamma = \frac{\omega_0^3 |\langle e|d|g \rangle|^2}{3\pi\epsilon_0 \hbar c^3}
\end{equation}
\end{definition}

\begin{definition}[Dipole Matrix Element]
The transition dipole strength:
\begin{equation}
d = \sqrt{\frac{3\epsilon_0 \hbar \Gamma \lambda^3}{8\pi}}
\end{equation}
\end{definition}

\subsection{D-Line Transitions}

\begin{definition}[D1 and D2 Lines]
The D-lines are transitions from the ground $nS_{1/2}$ state:
\begin{align}
\text{D1}: \quad nS_{1/2} &\rightarrow nP_{1/2} \\
\text{D2}: \quad nS_{1/2} &\rightarrow nP_{3/2}
\end{align}
\end{definition}

\begin{theorem}[Isotope Shift]
The isotope shift arises from:
\begin{equation}
\Delta E_{IS} = \Delta E_{mass} + \Delta E_{field}
\end{equation}
where mass shift is due to nuclear recoil and field shift is due to finite nuclear size.
\end{theorem}

\subsection{Dipole Strength Formula}

The code computes the reduced dipole matrix element:
\begin{lstlisting}
dipoleStrength=3.0*sqrt(e0*hbar*(2.0*NatGamma*(10.0**6))*(wavelength**3)/(8.0*pi))
\end{lstlisting}

This follows from inverting the spontaneous emission rate formula.

\section{Summary}
%==============================================================================

The \texttt{AtomConstants.py} module provides:

\begin{enumerate}
\item Atomic species classes: Nuclear spin, hyperfine constants, mass
\item Transition classes: Wavelength, linewidth, dipole strength
\item Isotope-transition classes: Excited state parameters, isotope shifts
\item Convenience dictionary for programmatic access
\end{enumerate}

Supported species: Rb-85, Rb-87, Cs-133, K-39, K-40, K-41, Na-23, IdealAtom.

All constants are in SI units except hyperfine constants (MHz) and masses (amu).
